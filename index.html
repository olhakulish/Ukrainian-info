    <script>

        // Use current URL for sharing

        const GAME_URL = window.location.href;

 

        const gameData = [

            {

                question: "What do the colors of the Ukrainian flag represent?",

                options: [

                    "The ocean and sandy beaches",

                    "Blue sky and golden wheat fields",

                    "A blue river and yellow sunflowers",

                    "Peace and bravery"

                ],

                correct: 1,

                fact: "The blue represents the sky, and the yellow represents the vast fields of wheat that cover much of the country."

            },

            {

                question: "What is the capital city of Ukraine, famous for its golden-domed churches?",

                options: ["Lviv", "Odesa", "Kharkiv", "Kyiv"],

                correct: 3,

                fact: "Kyiv is one of the oldest cities in Eastern Europe, founded over 1,500 years ago!"

            },

            {

                question: "Which traditional Ukrainian soup is made mainly from beetroots?",

                options: ["Goulash", "Borscht", "Solyanka", "Gazpacho"],

                correct: 1,

                fact: "Borscht is so important to Ukrainian identity that it is protected by UNESCO as intangible cultural heritage."

            },

            {

                question: "What is the national symbol of Ukraine (the 'Tryzub')?",

                options: ["A Double Eagle", "A Golden Lion", "A Trident", "A Rising Sun"],

                correct: 2,

                fact: "The Trident dates back over 1,000 years to the time of Prince Volodymyr the Great."

            },

            {

                question: "Ukraine is the world's largest producer of which product?",

                options: ["Olive Oil", "Coffee", "Maple Syrup", "Sunflower Oil"],

                correct: 3,

                fact: "Ukraine is the top exporter of sunflower oil. The sunflower is also the national flower and a symbol of peace."

            }

        ];

 

        let currentQuestion = 0;

        let score = 0;

        let canAnswer = true;

        let qrGenerated = false;

 

        const startScreen = document.getElementById('start-screen');

        const quizScreen = document.getElementById('quiz-screen');

        const endScreen = document.getElementById('end-screen');

        const questionText = document.getElementById('question-text');

        const optionsGrid = document.getElementById('options-grid');

        const feedback = document.getElementById('feedback');

        const nextBtn = document.getElementById('next-btn');

        const levelIndicator = document.getElementById('level-indicator');

        const progressBar = document.getElementById('progress');

 

        function startGame() {

            startScreen.classList.add('hidden');

            quizScreen.classList.remove('hidden');

            loadQuestion();

        }

 

        function loadQuestion() {

            const data = gameData[currentQuestion];

            canAnswer = true;

            nextBtn.classList.add('hidden');

            feedback.classList.add('hidden');

           

            levelIndicator.innerText = `LEVEL ${currentQuestion + 1}/${gameData.length}`;

            progressBar.style.width = `${((currentQuestion + 1) / gameData.length) * 100}%`;

           

            questionText.innerText = data.question;

            optionsGrid.innerHTML = '';

 

            data.options.forEach((option, index) => {

                const btn = document.createElement('button');

                btn.className = 'option-btn p-4 rounded-xl text-left font-medium hover:bg-gray-50 active:bg-gray-100 transition shadow-sm bg-white';

                btn.innerText = option;

                btn.onclick = () => checkAnswer(index, btn);

                optionsGrid.appendChild(btn);

            });

           

            // Scroll to top on mobile when question changes

            document.querySelector('.content-area').scrollTop = 0;

        }

 

        function checkAnswer(selectedIndex, btn) {

            if (!canAnswer) return;

            canAnswer = false;

 

            const data = gameData[currentQuestion];

            const buttons = optionsGrid.getElementsByTagName('button');

 

            if (selectedIndex === data.correct) {

                btn.classList.add('correct');

                score++;

                feedback.innerText = "Correct! ✨ " + data.fact;

                feedback.className = "mt-4 p-4 text-center font-bold text-sm md:text-lg text-green-700 bg-green-50 rounded-xl border border-green-100";

            } else {

                btn.classList.add('wrong');

                buttons[data.correct].classList.add('correct');

                feedback.innerText = "Not quite. ℹ️ " + data.fact;

                feedback.className = "mt-4 p-4 text-center font-bold text-sm md:text-lg text-red-700 bg-red-50 rounded-xl border border-red-100";

            }

 

            feedback.classList.remove('hidden');

            nextBtn.classList.remove('hidden');

           

            // Auto scroll to feedback on small screens

            setTimeout(() => {

                feedback.scrollIntoView({ behavior: 'smooth', block: 'nearest' });

            }, 100);

        }

 

        function nextQuestion() {

            currentQuestion++;

            if (currentQuestion < gameData.length) {

                loadQuestion();

            } else {

                showEndScreen();

            }

        }

 

        function showEndScreen() {

            quizScreen.classList.add('hidden');

            endScreen.classList.remove('hidden');

            document.getElementById('final-score').innerText = `Your Knowledge: ${score}/${gameData.length}`;

           

            const facts = [

                "Ukraine is the second largest country in Europe by land area!",

                "The world's deepest metro station, Arsenalna, is in Kyiv (105.5 meters deep).",

                "The world's longest musical instrument, the Trembita, comes from the Ukrainian Carpathians.",

                "Ukraine is home to 7 UNESCO World Heritage Sites."

            ];

            document.getElementById('random-fact').innerText = facts[Math.floor(Math.random() * facts.length)];

 

            if (!qrGenerated) {

                const qrContainer = document.getElementById("qrcode");

                document.getElementById("display-url").innerText = GAME_URL;

               

                new QRCode(qrContainer, {

                    text: GAME_URL,

                    width: 150,

                    height: 150,

                    colorDark : "#0057B7",

                    colorLight : "#ffffff",

                    correctLevel : QRCode.CorrectLevel.H

                });

                qrGenerated = true;

            }

            document.querySelector('.content-area').scrollTop = 0;

        }

 

        function restartGame() {

            currentQuestion = 0;

            score = 0;

            endScreen.classList.add('hidden');

            startScreen.classList.remove('hidden');

        }

    </script>
